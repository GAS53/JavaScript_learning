вызовы скриптов
    < script defer src = "https://javascript.info/article/script-async-defer/long.js?speed=1" ></script >
    <script async src="https://javascript.info/article/script-async-defer/long.js"></script>

defer // используются когда нужен доступ ко всему DOM или важен порядок выполнения
    defer - скрипты исполняются в порядке их расположения в документе
    defer - работает только для внешних скриптов - src=...
    выполняется когда DOM - дерево полностью построено
    не блокируют страницу

async // выполнить асинхронно, где порядок выполнения не играет роли
    страница не ждет асинхронных скриптов
    асинхронные скрипты не ждут друг друга
    другие скрипты не ждут асинхронные
    содержимое отображается сразу
    скрипты выполняются в порядке загрузки
